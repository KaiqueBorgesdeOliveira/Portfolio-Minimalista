Automação 
# Automacao operacional com Python
import subprocess

def health_check(hosts):
    for host in hosts:
        result = subprocess.run(["ping", "-c", "1", host], capture_output=True)
        print(host, "OK" if result.returncode == 0 else "FALHA")
